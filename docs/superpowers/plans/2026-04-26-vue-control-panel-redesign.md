# Vue Control Panel Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the hand-authored embedded control panel with a Vue/Vite-authored panel that is generated back into `ControlPanelHtml.cs` and keeps the plugin's localhost-only static delivery.

**Architecture:** Add a small Vue project under `src/HUnityAutoTranslator.ControlPanel`, build it as a single static HTML file with inlined CSS/JS, and generate `src/HUnityAutoTranslator.Plugin/Web/ControlPanelHtml.cs` from that output. Keep `LocalHttpServer` and the existing `/api/*` contracts stable while porting UI behavior into focused Vue components and composables.

**Tech Stack:** Vue 3, Vite, TypeScript, PowerShell generator script, existing .NET source tests, existing plugin HTTP APIs.

---

Create `build/generate-control-panel.ps1`:

```powershell
param(
    [string]$InputHtml = "",
    [string]$OutputFile = ""
)

$ErrorActionPreference = "Stop"

$root = Resolve-Path -LiteralPath (Join-Path $PSScriptRoot "..")
$defaultInput = Join-Path $root "src\HUnityAutoTranslator.ControlPanel\dist\index.html"
$defaultOutput = Join-Path $root "src\HUnityAutoTranslator.Plugin\Web\ControlPanelHtml.cs"

if ([string]::IsNullOrWhiteSpace($InputHtml)) {
    $InputHtml = $defaultInput
}

if ([string]::IsNullOrWhiteSpace($OutputFile)) {
    $OutputFile = $defaultOutput
}

if (-not (Test-Path -LiteralPath $InputHtml)) {
    throw "Built control panel HTML not found: $InputHtml"
}

$inputPath = Resolve-Path -LiteralPath $InputHtml
$distRoot = Split-Path -Parent $inputPath
$html = Get-Content -LiteralPath $inputPath -Raw -Encoding UTF8

function Read-DistAsset([string]$assetReference) {
    $assetPath = $assetReference.TrimStart(".", "/", "\").Replace("/", [System.IO.Path]::DirectorySeparatorChar)
    $fullPath = Join-Path $distRoot $assetPath
    if (-not (Test-Path -LiteralPath $fullPath)) {
        throw "Referenced control panel asset not found: $assetReference"
    }

    return Get-Content -LiteralPath $fullPath -Raw -Encoding UTF8
}

$html = [regex]::Replace(
    $html,
    '<link[^>]+rel="modulepreload"[^>]*>',
    '',
    [System.Text.RegularExpressions.RegexOptions]::IgnoreCase)

$html = [regex]::Replace(
    $html,
    '<link[^>]+rel="stylesheet"[^>]+href="([^"]+)"[^>]*>',
    {
        param($match)
        $css = Read-DistAsset $match.Groups[1].Value
        "<style>`n$css`n</style>"
    },
    [System.Text.RegularExpressions.RegexOptions]::IgnoreCase)

$html = [regex]::Replace(
    $html,
    '<script([^>]*)src="([^"]+)"([^>]*)></script>',
    {
        param($match)
        $before = $match.Groups[1].Value -replace '\s*crossorigin', ''
        $after = $match.Groups[3].Value -replace '\s*crossorigin', ''
        $js = Read-DistAsset $match.Groups[2].Value
        "<script$before$after>`n$js`n</script>"
    },
    [System.Text.RegularExpressions.RegexOptions]::IgnoreCase)

if ($html -match "https?://") {
    throw "Generated control panel must not reference remote assets."
}

$delimiter = '""""""""'
$content = @(
    'namespace HUnityAutoTranslator.Plugin;'
    ''
    'internal static class ControlPanelHtml'
    '{'
    '    // <auto-generated by build/generate-control-panel.ps1>'
    "    public const string Document = $delimiter"
    $html
    "$delimiter;"
    '}'
) -join "`n"

$utf8NoBom = New-Object System.Text.UTF8Encoding($false)
[System.IO.File]::WriteAllText($OutputFile, $content, $utf8NoBom)
Write-Host "Generated control panel: $OutputFile"
```

import type { JsonData } from "../types/api";

export async function api<T>(path: string, options: RequestInit = {}): Promise<T> {
  const response = await fetch(path, {
    ...options,
    headers: {
      "Content-Type": "application/json",
      ...(options.headers ?? {})
    }
  });

  if (!response.ok) {
    const text = await response.text();
    throw new Error(text || `HTTP ${response.status}`);
  }

  const contentType = response.headers.get("content-type") ?? "";
  if (contentType.includes("application/json")) {
    return await response.json() as T;
  }

  return await response.text() as T;
}

export function postJson<T>(path: string, body: JsonData): Promise<T> {
  return api<T>(path, { method: "POST", body: JSON.stringify(body) });
}

export function patchJson<T>(path: string, body: JsonData): Promise<T> {
  return api<T>(path, { method: "PATCH", body: JSON.stringify(body) });
}

export function deleteJson<T>(path: string, body: JsonData): Promise<T> {
  return api<T>(path, { method: "DELETE", body: JSON.stringify(body) });
}

import { computed, reactive } from "vue";
import { api, postJson } from "../api/client";
import type { ConnectionState, ControlPanelState, JsonData, PageKey, RecentTranslation, ToastMessage, ToastKind } from "../types/api";

const themeOrder = ["system", "light", "dark"] as const;
type ThemeChoice = typeof themeOrder[number];

export const store = reactive({
  activePage: "status" as PageKey,
  connection: "reconnecting" as ConnectionState,
  state: {} as ControlPanelState,
  configDirty: false,
  theme: loadTheme(),
  toasts: [] as ToastMessage[],
  nextToastId: 1
});

export const pageTitle = computed(() => {
  const titles: Record<PageKey, string> = {
    status: "运行状态",
    plugin: "插件设置",
    ai: "AI 翻译设置",
    glossary: "术语库",
    editor: "文本编辑",
    about: "版本信息"
  };
  return titles[store.activePage];
});

function loadTheme(): ThemeChoice {
  const value = localStorage.getItem("hunity.theme");
  return value === "light" || value === "dark" || value === "system" ? value : "system";
}

export function applyTheme(): void {
  if (store.theme === "system") {
    document.documentElement.removeAttribute("data-theme");
  } else {
    document.documentElement.dataset.theme = store.theme;
  }
}

export function cycleTheme(): void {
  const index = themeOrder.indexOf(store.theme);
  store.theme = themeOrder[(index + 1) % themeOrder.length];
  localStorage.setItem("hunity.theme", store.theme);
  applyTheme();
}

export function showToast(text: string, kind: ToastKind = "info"): void {
  const toast: ToastMessage = { id: store.nextToastId++, kind, text };
  store.toasts.push(toast);
  window.setTimeout(() => dismissToast(toast.id), 4200);
}

export function dismissToast(id: number): void {
  const index = store.toasts.findIndex(toast => toast.id === id);
  if (index >= 0) {
    store.toasts.splice(index, 1);
  }
}

export async function refreshState(options: { forceConfigFields?: boolean } = {}): Promise<void> {
  try {
    const nextState = await api<ControlPanelState>("/api/state");
    store.state = nextState;
    store.connection = "connected";
    if (options.forceConfigFields) {
      store.configDirty = false;
    }
  } catch (error) {
    store.connection = "disconnected";
    store.state = {
      ...store.state,
      Enabled: false,
      LastError: error instanceof Error ? error.message : "控制面板连接中断"
    };
  }
}

export async function saveConfig(config: Record<string, JsonData | RecentTranslation[] | undefined>): Promise<void> {
  store.state = await postJson<ControlPanelState>("/api/config", config as JsonData);
  store.configDirty = false;
  showToast("设置已保存", "ok");
}

import type { TranslationCacheEntry } from "../types/api";

export interface TableColumn {
  key: keyof TranslationCacheEntry & string;
  label: string;
  editable: boolean;
  width: number;
}

export const defaultColumns: TableColumn[] = [
  { key: "SourceText", label: "原文", editable: false, width: 260 },
  { key: "TranslatedText", label: "译文", editable: true, width: 260 },
  { key: "TargetLanguage", label: "目标语言", editable: false, width: 110 },
  { key: "SceneName", label: "场景", editable: true, width: 150 },
  { key: "ComponentHierarchy", label: "组件层级", editable: true, width: 240 },
  { key: "ComponentType", label: "组件类型", editable: true, width: 130 },
  { key: "ProviderKind", label: "服务商", editable: false, width: 110 },
  { key: "ProviderModel", label: "模型", editable: false, width: 150 },
  { key: "CreatedUtc", label: "创建时间", editable: false, width: 170 },
  { key: "UpdatedUtc", label: "更新时间", editable: false, width: 170 }
];

export function rowKey(row: TranslationCacheEntry): string {
  return [
    row.SourceText ?? "",
    row.TargetLanguage ?? "",
    row.SceneName ?? "",
    row.ComponentHierarchy ?? "",
    row.ComponentType ?? ""
  ].join("\u001f");
}

export function loadVisibleColumns(): string[] {
  const raw = localStorage.getItem("hunity.editor.visibleColumns");
  if (!raw) return defaultColumns.map(column => column.key);
  try {
    const parsed = JSON.parse(raw);
    return Array.isArray(parsed) ? parsed.filter(value => typeof value === "string") : defaultColumns.map(column => column.key);
  } catch {
    return defaultColumns.map(column => column.key);
  }
}

export function saveVisibleColumns(keys: string[]): void {
  localStorage.setItem("hunity.editor.visibleColumns", JSON.stringify(keys));
}

export function loadColumnOrder(): string[] {
  const raw = localStorage.getItem("hunity.editor.columnOrder");
  if (!raw) return defaultColumns.map(column => column.key);
  try {
    const parsed = JSON.parse(raw);
    return Array.isArray(parsed) ? parsed.filter(value => typeof value === "string") : defaultColumns.map(column => column.key);
  } catch {
    return defaultColumns.map(column => column.key);
  }
}

export function saveColumnOrder(keys: string[]): void {
  localStorage.setItem("hunity.editor.columnOrder", JSON.stringify(keys));
}

export function toTsv(rows: TranslationCacheEntry[], columns: TableColumn[]): string {
  return rows.map(row => columns.map(column => String(row[column.key] ?? "").replace(/\r?\n/g, " ")).join("\t")).join("\n");
}
